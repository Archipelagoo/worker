// ABSEN ARCHIPELAGO - Google Apps Script
const SPREADSHEET_ID = '1FllBcwBjTr_jfPylhyWwHt6ZE16j6fJpSa0C9hYEw44';
const SHEET_WORKERS = 'Form responses 1';
const SHEET_ABSEN = 'Absen recap';
const FOLDER_PHOTOS = '1CJ46h8M89mzc3XKEg-4hJPV-2voMN3OD';

const COLORS = {
  'Start Work': '#22c55e',
  'Finish Work': '#3b82f6',
  'Sakit': '#f97316',
  'Standby': '#a855f7'
};

function doGet(e) {
  const action = e.parameter.action;
  try {
    if (action === 'getWorkers') return jsonResponse(getWorkers());
    if (action === 'getAttendance') return jsonResponse(getAttendance(e.parameter.date));
    if (action === 'getPhotos') return jsonResponse(getPhotos(e.parameter.date));
    if (action === 'getPhotoFolders') return jsonResponse(getPhotoFolders());
    return jsonResponse({ error: 'Invalid action' });
  } catch (error) { return jsonResponse({ error: error.message }); }
}

function doPost(e) {
  try {
    let data;
    if (e.parameter && e.parameter.data) {
      data = JSON.parse(e.parameter.data);
    } else if (e.postData && e.postData.contents) {
      data = JSON.parse(e.postData.contents);
    } else {
      return jsonResponse({ error: 'No data received' });
    }
    if (data.action === 'submitAttendance') return jsonResponse(submitAttendance(data));
    if (data.action === 'uploadPhoto') return jsonResponse(uploadPhoto(data));
    return jsonResponse({ error: 'Invalid action' });
  } catch (error) { return jsonResponse({ error: error.message }); }
}

function jsonResponse(data) {
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}

function getWorkers() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName(SHEET_WORKERS);
  if (!sheet) return { success: false, error: 'Sheet tidak ditemukan!', workers: [] };
  const data = sheet.getDataRange().getValues();
  const workers = [];
  for (let i = 1; i < data.length; i++) {
    const nama = data[i][1], foto = data[i][2];
    if (nama && nama.toString().trim() !== '') {
      workers.push({ id: i, name: nama.toString().trim(), photo: foto ? foto.toString().trim() : '' });
    }
  }
  return { success: true, workers: workers };
}

function syncWorkersToAbsenSheet() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const workersSheet = ss.getSheetByName(SHEET_WORKERS);
  let absenSheet = ss.getSheetByName(SHEET_ABSEN);
  
  if (!absenSheet) {
    absenSheet = ss.insertSheet(SHEET_ABSEN);
    absenSheet.getRange('A1').setValue('Tanggal').setBackground('#1a2d4a').setFontColor('#ffffff').setFontWeight('bold');
    absenSheet.getRange('A2').setValue('Jam').setBackground('#1a2d4a').setFontColor('#ffffff').setFontWeight('bold');
  }
  
  const workersData = workersSheet.getDataRange().getValues();
  const currentWorkers = [];
  for (let i = 1; i < workersData.length; i++) {
    const nama = workersData[i][1];
    if (nama && nama.toString().trim() !== '') {
      currentWorkers.push(nama.toString().trim());
    }
  }
  
  const lastRow = absenSheet.getLastRow();
  const existingWorkers = [];
  if (lastRow >= 3) {
    const existingData = absenSheet.getRange(3, 1, lastRow - 2, 1).getValues();
    for (let i = 0; i < existingData.length; i++) {
      if (existingData[i][0]) existingWorkers.push(existingData[i][0].toString().trim());
    }
  }
  
  let nextRow = lastRow >= 3 ? lastRow + 1 : 3;
  currentWorkers.forEach(worker => {
    if (!existingWorkers.includes(worker)) {
      absenSheet.getRange(nextRow, 1).setValue(worker).setBackground('#1a2d4a').setFontColor('#ffffff').setFontWeight('bold');
      nextRow++;
    }
  });
  
  return currentWorkers;
}

function submitAttendance(data) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = ss.getSheetByName(SHEET_ABSEN);
  
  syncWorkersToAbsenSheet();
  sheet = ss.getSheetByName(SHEET_ABSEN);
  
  const lastCol = sheet.getLastColumn();
  const newCol = lastCol + 1;
  
  sheet.getRange(1, newCol).setValue(data.tanggal);
  sheet.getRange(2, newCol).setValue(data.jam);
  
  const lastRow = sheet.getLastRow();
  if (lastRow >= 3) {
    const workerNames = sheet.getRange(3, 1, lastRow - 2, 1).getValues();
    
    const statusMap = {};
    data.attendance.forEach(item => {
      statusMap[item.name.trim()] = item.status;
    });
    
    for (let i = 0; i < workerNames.length; i++) {
      const workerName = workerNames[i][0].toString().trim();
      const status = statusMap[workerName] || '-';
      const cell = sheet.getRange(3 + i, newCol);
      cell.setValue(status);
      if (COLORS[status]) {
        cell.setBackground(COLORS[status]).setFontColor('#ffffff');
      }
    }
  }
  
  return { success: true, message: 'Absensi berhasil disimpan!' };
}

function getAttendance(date) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName(SHEET_ABSEN);
  if (!sheet) return { success: true, data: [] };
  
  const allData = sheet.getDataRange().getValues();
  if (allData.length < 3) return { success: true, data: [] };
  
  const results = [];
  
  for (let col = 1; col < allData[0].length; col++) {
    const tanggal = allData[0][col];
    if (tanggal === date) {
      const jam = allData[1][col];
      const workers = [];
      
      for (let row = 2; row < allData.length; row++) {
        const workerName = allData[row][0];
        if (workerName) {
          workers.push({
            name: workerName.toString().trim(),
            status: allData[row][col] || '-'
          });
        }
      }
      
      results.push({ tanggal: tanggal, jam: jam, workers: workers });
    }
  }
  
  return { success: true, data: results };
}

function uploadPhoto(data) {
  const parentFolder = DriveApp.getFolderById(FOLDER_PHOTOS);
  const folderName = data.tanggal.replace(/\//g, '-');
  
  let dateFolder;
  const folders = parentFolder.getFoldersByName(folderName);
  if (folders.hasNext()) {
    dateFolder = folders.next();
  } else {
    dateFolder = parentFolder.createFolder(folderName);
  }
  
  const base64Data = data.photo.replace(/^data:image\/(png|jpeg|jpg);base64,/, '');
  const blob = Utilities.newBlob(Utilities.base64Decode(base64Data), 'image/jpeg', 'foto_' + data.jam.replace(/:/g, '-') + '_' + data.index + '.jpg');
  dateFolder.createFile(blob);
  
  return { success: true };
}

function getPhotoFolders() {
  try {
    const parentFolder = DriveApp.getFolderById(FOLDER_PHOTOS);
    const folders = parentFolder.getFolders();
    const folderList = [];
    
    while (folders.hasNext()) {
      const folder = folders.next();
      folderList.push(folder.getName());
    }
    
    folderList.sort((a, b) => {
      const dateA = a.split('-').reverse().join('-');
      const dateB = b.split('-').reverse().join('-');
      return dateB.localeCompare(dateA);
    });
    
    return { success: true, folders: folderList };
  } catch (error) {
    return { success: false, error: error.message, folders: [] };
  }
}

function getPhotos(date) {
  try {
    const parentFolder = DriveApp.getFolderById(FOLDER_PHOTOS);
    const folderName = date.replace(/\//g, '-');
    const folders = parentFolder.getFoldersByName(folderName);
    
    if (!folders.hasNext()) {
      return { success: true, photos: [] };
    }
    
    const dateFolder = folders.next();
    const files = dateFolder.getFiles();
    const photos = [];
    
    while (files.hasNext()) {
      const file = files.next();
      if (file.getMimeType().startsWith('image/')) {
        photos.push({
          name: file.getName(),
          url: 'https://lh3.googleusercontent.com/d/' + file.getId(),
          id: file.getId()
        });
      }
    }
    
    return { success: true, photos: photos };
  } catch (error) {
    return { success: false, error: error.message, photos: [] };
  }
}
