const SPREADSHEET_ID = '1FllBcwBjTr_jfPylhyWwHt6ZE16j6fJpSa0C9hYEw44';
const PHOTO_FOLDER_ID = '1CJ46h8M89mzc3XKEg-4hJPV-2voMN3OD';

const BULAN_INDO = ['Januari','Februari','Maret','April','Mei','Juni','Juli','Agustus','September','Oktober','November','Desember'];

// Warna status (solid)
const STATUS_COLORS = {
  'Start Work': '#22c55e',
  'Finish Work': '#3b82f6',
  'Sakit': '#f97316',
  'Standby': '#a855f7',
  '-': '#ef4444'
};

function doGet(e) {
  const action = e.parameter.action;
  let result;
  
  try {
    switch(action) {
      case 'getWorkers':
        result = getWorkers();
        break;
      case 'getAttendance':
        result = getAttendance(e.parameter.date);
        break;
      case 'getMonthData':
        result = getMonthData(e.parameter.month, e.parameter.year);
        break;
      case 'getPhotoFolders':
        result = getPhotoFolders();
        break;
      case 'getPhotos':
        result = getPhotos(e.parameter.date);
        break;
      default:
        result = { success: false, error: 'Unknown action' };
    }
  } catch(err) {
    result = { success: false, error: err.toString() };
  }
  
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    const data = JSON.parse(e.parameter.data);
    let result;
    
    switch(data.action) {
      case 'submitAttendance':
        result = submitAttendance(data);
        break;
      case 'uploadPhotos':
        result = uploadPhotos(data);
        break;
      default:
        result = { success: false, error: 'Unknown action' };
    }
    
    return ContentService.createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ 
      success: false, 
      error: err.toString() 
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

function getWorkers() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName('Form Responses 1');
  if (!sheet) return { success: true, workers: [] };
  
  const data = sheet.getDataRange().getValues();
  const workers = [];
  
  for (let i = 1; i < data.length; i++) {
    const nama = data[i][1];
    const photo = data[i][2] || '';
    
    if (nama && nama.toString().trim() !== '') {
      workers.push({
        id: i,
        name: nama.toString().trim(),
        photo: photo.toString().trim()
      });
    }
  }
  
  workers.sort((a, b) => a.name.localeCompare(b.name));
  return { success: true, workers };
}

function submitAttendance(data) {
  const { tanggal, jam, attendance } = data;
  
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = ss.getSheetByName('Absen Recap');
  
  if (!sheet) {
    sheet = ss.insertSheet('Absen Recap');
    sheet.getRange(1, 1).setValue('Tanggal');
    sheet.getRange(2, 1).setValue('Jam');
  }
  
  // Sync nama dari Form Responses 1
  syncWorkerNames(sheet);
  
  const lastCol = sheet.getLastColumn();
  const newCol = lastCol < 1 ? 2 : lastCol + 1;
  
  // Tulis tanggal dan jam
  sheet.getRange(1, newCol).setValue(tanggal);
  sheet.getRange(2, newCol).setValue(jam);
  
  const lastRow = sheet.getLastRow();
  if (lastRow < 3) return { success: true };
  
  // Ambil semua nama di kolom A (mulai baris 3)
  const workerNames = sheet.getRange(3, 1, lastRow - 2, 1).getValues().flat().map(n => n.toString().trim());
  
  // Tulis status untuk setiap pekerja
  attendance.forEach(a => {
    const cleanName = a.name.toString().trim();
    const rowIndex = workerNames.findIndex(n => n === cleanName);
    
    if (rowIndex !== -1) {
      const cell = sheet.getRange(rowIndex + 3, newCol);
      const status = a.status || '-';
      cell.setValue(status);
      
      // Terapkan warna
      const color = STATUS_COLORS[status] || STATUS_COLORS['-'];
      cell.setBackground(color);
      cell.setFontColor('#ffffff');
      cell.setFontWeight('bold');
      cell.setHorizontalAlignment('center');
    }
  });
  
  // Terapkan warna tanggal
  applyDateRowColors(sheet);
  
  return { success: true };
}

function syncWorkerNames(recapSheet) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const formSheet = ss.getSheetByName('Form Responses 1');
  if (!formSheet) return;
  
  const formData = formSheet.getDataRange().getValues();
  const newNames = [];
  for (let i = 1; i < formData.length; i++) {
    const nama = formData[i][1];
    if (nama && nama.toString().trim() !== '') {
      newNames.push(nama.toString().trim());
    }
  }
  newNames.sort((a, b) => a.localeCompare(b, 'id'));
  
  if (recapSheet.getRange(1, 1).getValue() === '') {
    recapSheet.getRange(1, 1).setValue('Tanggal');
    recapSheet.getRange(2, 1).setValue('Jam');
  }
  
  const lastRow = recapSheet.getLastRow();
  let existingNames = [];
  if (lastRow >= 3) {
    existingNames = recapSheet.getRange(3, 1, lastRow - 2, 1).getValues().flat().map(n => n.toString().trim()).filter(n => n !== '');
  }
  
  newNames.forEach(name => {
    if (!existingNames.includes(name)) {
      const nextRow = recapSheet.getLastRow() + 1;
      recapSheet.getRange(nextRow, 1).setValue(name);
      existingNames.push(name);
    }
  });
}

function applyDateRowColors(sheet) {
  const lastCol = sheet.getLastColumn();
  if (lastCol < 2) return;
  
  const dates = sheet.getRange(1, 2, 1, lastCol - 1).getValues()[0];
  
  let currentDate = '';
  let colorIndex = 0;
  const grayColor = '#6b7280';
  const whiteColor = '#ffffff';
  
  for (let col = 2; col <= lastCol; col++) {
    const dateVal = dates[col - 2];
    if (dateVal !== currentDate) {
      currentDate = dateVal;
      colorIndex++;
    }
    
    const bgColor = colorIndex % 2 === 1 ? grayColor : whiteColor;
    const textColor = colorIndex % 2 === 1 ? '#ffffff' : '#000000';
    
    const dateCell = sheet.getRange(1, col);
    dateCell.setBackground(bgColor);
    dateCell.setFontColor(textColor);
    dateCell.setFontWeight('bold');
    dateCell.setHorizontalAlignment('center');
    
    const jamCell = sheet.getRange(2, col);
    jamCell.setBackground('#1e293b');
    jamCell.setFontColor('#fbbf24');
    jamCell.setFontWeight('bold');
    jamCell.setHorizontalAlignment('center');
  }
}

function getAttendance(date) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName('Absen Recap');
  if (!sheet) return { success: true, columns: [], workers: [] };
  
  const lastCol = sheet.getLastColumn();
  const lastRow = sheet.getLastRow();
  if (lastCol < 2 || lastRow < 3) return { success: true, columns: [], workers: [] };
  
  const allData = sheet.getRange(1, 1, lastRow, lastCol).getValues();
  
  const matchingCols = [];
  for (let col = 1; col < lastCol; col++) {
    const cellDate = allData[0][col] ? allData[0][col].toString() : '';
    if (cellDate === date) {
      matchingCols.push({
        colIndex: col,
        tanggal: cellDate,
        jam: allData[1][col] ? allData[1][col].toString() : ''
      });
    }
  }
  
  if (matchingCols.length === 0) {
    return { success: true, columns: [], workers: [] };
  }
  
  const workers = [];
  for (let row = 2; row < lastRow; row++) {
    const name = allData[row][0];
    if (name && name.toString().trim() !== '') {
      const statuses = {};
      matchingCols.forEach(mc => {
        const statusVal = allData[row][mc.colIndex];
        statuses[mc.jam] = statusVal ? statusVal.toString() : '-';
      });
      workers.push({ name: name.toString().trim(), statuses });
    }
  }
  
  // Sort workers by name
  workers.sort((a, b) => a.name.localeCompare(b.name, 'id'));
  
  return { 
    success: true, 
    columns: matchingCols.map(mc => ({ tanggal: mc.tanggal, jam: mc.jam })),
    workers 
  };
}

function getMonthData(month, year) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName('Absen Recap');
  if (!sheet) return { success: true, data: {} };
  
  const lastCol = sheet.getLastColumn();
  if (lastCol < 2) return { success: true, data: {} };
  
  const dates = sheet.getRange(1, 2, 1, lastCol - 1).getValues()[0];
  const result = {};
  
  dates.forEach(dateStr => {
    if (!dateStr) return;
    const parsed = parseTanggalIndo(dateStr.toString());
    if (parsed && parsed.month == month && parsed.year == year) {
      result[parsed.day] = (result[parsed.day] || 0) + 1;
    }
  });
  
  return { success: true, data: result };
}

function parseTanggalIndo(str) {
  if (!str) return null;
  const parts = str.toString().split(' ');
  if (parts.length !== 3) return null;
  
  const day = parseInt(parts[0]);
  const monthIndex = BULAN_INDO.indexOf(parts[1]) + 1;
  const year = parseInt(parts[2]);
  
  if (isNaN(day) || monthIndex === 0 || isNaN(year)) return null;
  return { day, month: monthIndex, year };
}

function uploadPhotos(data) {
  const { tanggal, jam, photos } = data;
  
  if (!photos || photos.length === 0) {
    return { success: true };
  }
  
  const folder = getOrCreateDateFolder(tanggal);
  const timeStr = jam.replace(/:/g, '-');
  
  photos.forEach((photoBase64, index) => {
    try {
      const base64Data = photoBase64.split(',')[1];
      const blob = Utilities.newBlob(
        Utilities.base64Decode(base64Data), 
        'image/jpeg', 
        `Absen_${timeStr}_${index + 1}.jpg`
      );
      folder.createFile(blob);
    } catch(err) {
      console.error('Error saving photo ' + (index + 1) + ': ' + err);
    }
  });
  
  return { success: true };
}

function getOrCreateDateFolder(tanggal) {
  const parentFolder = DriveApp.getFolderById(PHOTO_FOLDER_ID);
  const folderName = tanggal.replace(/ /g, '_');
  
  const folders = parentFolder.getFoldersByName(folderName);
  if (folders.hasNext()) {
    return folders.next();
  }
  return parentFolder.createFolder(folderName);
}

function getPhotoFolders() {
  try {
    const folder = DriveApp.getFolderById(PHOTO_FOLDER_ID);
    const subFolders = folder.getFolders();
    const folders = [];
    
    while (subFolders.hasNext()) {
      const f = subFolders.next();
      folders.push(f.getName());
    }
    
    folders.sort((a, b) => {
      const parseDate = (str) => {
        const p = str.replace(/_/g, ' ').split(' ');
        if (p.length !== 3) return new Date(0);
        const day = parseInt(p[0]);
        const monthIndex = BULAN_INDO.indexOf(p[1]);
        const year = parseInt(p[2]);
        return new Date(year, monthIndex, day);
      };
      return parseDate(b) - parseDate(a);
    });
    
    return { success: true, folders };
  } catch(err) {
    return { success: false, error: err.toString() };
  }
}

function getPhotos(date) {
  try {
    const folder = DriveApp.getFolderById(PHOTO_FOLDER_ID);
    const folderName = date.replace(/ /g, '_');
    const subFolders = folder.getFoldersByName(folderName);
    
    if (!subFolders.hasNext()) {
      return { success: true, photos: [] };
    }
    
    const dateFolder = subFolders.next();
    const files = dateFolder.getFiles();
    const photos = [];
    
    while (files.hasNext()) {
      const file = files.next();
      if (file.getMimeType().startsWith('image/')) {
        photos.push({
          name: file.getName(),
          url: 'https://lh3.googleusercontent.com/d/' + file.getId()
        });
      }
    }
    
    photos.sort((a, b) => a.name.localeCompare(b.name));
    return { success: true, photos };
  } catch(err) {
    return { success: false, error: err.toString() };
  }
}

// Sortir nama pekerja berdasarkan abjad dan rapikan
function sortAndFixRecapSheet() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName('Absen Recap');
  if (!sheet) return { success: false, error: 'Sheet Absen Recap tidak ditemukan' };
  
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  
  if (lastRow < 3 || lastCol < 1) return { success: false, error: 'Data tidak cukup' };
  
  // Ambil semua data
  const allData = sheet.getRange(1, 1, lastRow, lastCol).getValues();
  
  // Baris 1 = Tanggal, Baris 2 = Jam
  const headerRow1 = allData[0];
  const headerRow2 = allData[1];
  
  // Ambil data pekerja (mulai baris 3)
  const workerData = [];
  for (let i = 2; i < lastRow; i++) {
    const name = allData[i][0];
    if (name && name.toString().trim() !== '') {
      const rowData = {
        name: name.toString().trim(),
        statuses: allData[i].slice(1)
      };
      workerData.push(rowData);
    }
  }
  
  // Sortir berdasarkan nama (abjad Indonesia)
  workerData.sort((a, b) => a.name.localeCompare(b.name, 'id'));
  
  // Hapus data lama (baris 3 ke bawah)
  if (lastRow > 2) {
    sheet.getRange(3, 1, lastRow - 2, lastCol).clear();
  }
  
  // Tulis ulang data yang sudah diurutkan
  workerData.forEach((worker, index) => {
    const row = index + 3;
    
    // Tulis nama
    const nameCell = sheet.getRange(row, 1);
    nameCell.setValue(worker.name);
    nameCell.setBackground('#0f172a');
    nameCell.setFontColor('#ffffff');
    nameCell.setFontWeight('bold');
    
    // Tulis status
    worker.statuses.forEach((status, colIndex) => {
      const col = colIndex + 2;
      const cell = sheet.getRange(row, col);
      
      if (status && status.toString().trim() !== '') {
        const statusStr = status.toString().trim();
        cell.setValue(statusStr);
        
        // Terapkan warna
        const color = STATUS_COLORS[statusStr] || STATUS_COLORS['-'];
        cell.setBackground(color);
        cell.setFontColor('#ffffff');
        cell.setFontWeight('bold');
        cell.setHorizontalAlignment('center');
      }
    });
  });
  
  // Format header
  sheet.getRange(1, 1).setValue('Tanggal');
  sheet.getRange(2, 1).setValue('Jam');
  sheet.getRange(1, 1, 2, 1).setBackground('#1e293b');
  sheet.getRange(1, 1, 2, 1).setFontColor('#ffffff');
  sheet.getRange(1, 1, 2, 1).setFontWeight('bold');
  
  // Terapkan warna tanggal
  applyDateRowColors(sheet);
  
  return { success: true, message: 'Total ' + workerData.length + ' pekerja diurutkan!' };
}

// Terapkan warna ke SEMUA sel status
function applyColorsToAllData() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName('Absen Recap');
  if (!sheet) return { success: false, error: 'Sheet tidak ditemukan' };
  
  const lastRow = sheet.getLastRow();
  const lastCol = sheet.getLastColumn();
  
  if (lastRow < 3 || lastCol < 2) return { success: false, error: 'Data tidak cukup' };
  
  let coloredCount = 0;
  
  // Loop semua sel status (mulai baris 3, kolom 2)
  for (let row = 3; row <= lastRow; row++) {
    for (let col = 2; col <= lastCol; col++) {
      const cell = sheet.getRange(row, col);
      const value = cell.getValue();
      
      if (value && value.toString().trim() !== '') {
        const status = value.toString().trim();
        const color = STATUS_COLORS[status] || STATUS_COLORS['-'];
        
        cell.setBackground(color);
        cell.setFontColor('#ffffff');
        cell.setFontWeight('bold');
        cell.setHorizontalAlignment('center');
        coloredCount++;
      }
    }
  }
  
  // Terapkan warna tanggal
  applyDateRowColors(sheet);
  
  return { success: true, message: coloredCount + ' sel status diwarnai!' };
}
